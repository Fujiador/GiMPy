  8  This code solves an integer program by using an LP-based branch and bound 
  9  algorithm. The search strategy is controlled by the priority given to nodes in 
 10  the queue and there are two different rules available for solecting the 
 11  branching variable (most fractional and fixed). The complete_enumeration 
 12  variable can be used to turn off fathoming by bound. 
 13

 83      CONSTRAINTS = ["C"+str(i) for i in range(numCons)] 
 84      if layout == 'ladot': 
 85          VARIABLES = ["x_{"+str(i)+"}" for i in range(numVars)] 
 86      else: 
 87          VARIABLES = ["x"+str(i) for i in range(numVars)] 
 88      OBJ = {i : randint(1, maxObjCoeff) for i in VARIABLES} 
 89      MAT = {i : [randint(1, maxConsCoeff) if random() <= density else 0 
 90                  for j in CONSTRAINTS] for i in VARIABLES} 
 91      RHS = [randint(int(numVars*density*maxConsCoeff/2), int(numVars*density*maxConsCoeff/1.5)) for i in CONSTRAINTS]

 93  var   = LpVariable.dicts("", VARIABLES, 0, 1)

109  # 1 to force complete enumeration of nodes (no fathoming by bounding)

122  iter_count = 0

128  print "==========================================="

131  if branch_strategy == MOST_FRAC:

133  elif branch_strategy == FIXED: 
134      print "Fixed order"

139      print "Depth first search strategy"

143      print "Unknown search strategy %s" %search_strategy

157  Q.push((0, None, None, None, None), INFINITY)

159  # Branch and Bound Loop 
160  while not Q.isEmpty(): 
161   
162      cur_index, parent, branch_var, sense, rhs = Q.pop() 
163      if cur_index is not 0: 
164          cur_depth = T.get_node_attr(parent, 'level') + 1 
165      else: 
166          cur_depth = 0

168      print "" 
169      print "----------------------------------------------------" 
170      print "" 
171      print "Node: %s, Depth: %s, LB: %s" %(cur_index,cur_depth,LB) 
172   
173      #==================================== 
174      #    LP Relaxation 
175      #==================================== 
176      #Compute lower bound by LP relaxation 
177      prob = LpProblem("relax",LpMaximize) 
178      prob += lpSum([OBJ[i]*var[i] for i in VARIABLES]), "Objective" 
179      for j in range(numCons): 
180          prob += lpSum([MAT[i][j]*var[i] for i in VARIABLES]) <= RHS[j], \ 
181              CONSTRAINTS[j] 
182   
183      #Fix all prescribed variables 
184      branch_vars = [] 
185      if cur_index is not 0: 
186          print "Branching variables: " 
187          branch_vars.append(branch_var) 
188          if sense == '>=': 
189              prob += LpConstraint(lpSum(var[branch_var]) >= rhs) 
190          else: 
191              prob += LpConstraint(lpSum(var[branch_var]) <= rhs) 
192          print branch_var, 
193          pred = parent 
194          while str(pred) is not '0': 
195              pred_branch_var = T.get_node_attr(pred, 'branch_var') 
196              pred_rhs = T.get_node_attr(pred, 'rhs') 
197              pred_sense = T.get_node_attr(pred, 'sense') 
198              if pred_sense == '<=': 
199                  prob += LpConstraint(lpSum(var[pred_branch_var]) <= pred_rhs) 
200              else: 
201                  prob += LpConstraint(lpSum(var[pred_branch_var]) >= pred_rhs) 
202              print pred_branch_var, 
203              branch_vars.append(pred_branch_var) 
204              pred = T.get_node_attr(pred, 'parent')

206      print ""

209      prob.solve()

211      lp_count = lp_count +1

218          print "LP Solved, status: Infeasible" 
219      else: 
220          print "LP Solved, status: %s, obj: %s" %(LpStatus[prob.status], 
221                                                   value(prob.objective)) 
222   
223   
224      if(LpStatus[prob.status] == "Optimal"): 
225          relax = value(prob.objective) 
226      else: 
227          relax = INFINITY 
228   
229      integer_solution = 0 
230   
231      if (LpStatus[prob.status] == "Optimal"): 
232          var_values = dict([(i, var[i].varValue) for i in VARIABLES]) 
233          integer_solution = 1 
234          for i in VARIABLES: 
235              if (var_values[i] not in set([0,1])): 
236                  integer_solution = 0 
237                  break 
238          if (integer_solution and relax>LB): 
239              LB = relax 
240              for i in VARIABLES: 
241                  #these two have different data structures first one list 
242                  #second one dictionary 
243                  opt[i] = var_values[i] 
244              print "New best solution found, objective: %s" %relax 
245              for i in VARIABLES: 
246                  if var_values[i] > 0: 
247                      print "%s = %s" %(i, var_values[i]) 
248          elif (integer_solution and relax<=LB): 
249              print "New integer solution found, objective: %s" %relax 
250              for i in VARIABLES: 
251                  if var_values[i] > 0: 
252                      print "%s = %s" %(i, var_values[i]) 
253          else: 
254              print "Fractional solution:" 
255              for i in VARIABLES: 
256                  if var_values[i] > 0: 
257                      print "%s = %s" %(i, var_values[i])

259      #For complete enumeration

261          relax = LB - 1 
262   
263      if integer_solution: 
264          print "Integer solution" 
265          status = 'S' 
266          BAKstatus = 'integer' 
267          color = 'lightblue' 
268      elif infeasible: 
269          print "Infeasible node"

271          BAKstatus = 'infeasible' 
272          color = 'orange' 
273      elif not complete_enumeration and relax <= LB: 
274          print "Node pruned by bound (obj: %s, UB: %s)" %(relax,LB)

276          BAKstatus = 'fathomed' 
277          color = 'red'

319                      T.set_edge_attr(parent, cur_index, 'label',

322                      T.set_edge_attr(parent, cur_index, 'label',

325                  T.set_edge_attr(parent, cur_index, 'label',

342      if status == 'C': 
343   
344          # Branching: 
345          # Choose a variable for branching 
346          branching_var = -1 
347          if branch_strategy == FIXED: 
348              #fixed order 
349              for i in VARIABLES: 
350                  frac = min(var[i].varValue-math.floor(var[i].varValue), 
351                             math.ceil(var[i].varValue) - var[i].varValue) 
352                  if (frac > 0): 
353                      min_frac = frac 
354                      branching_var = i 
355                      break 
356          elif branch_strategy == MOST_FRAC: 
357              #most fractional variable 
358              min_frac = -1 
359              for i in VARIABLES: 
360                  frac = min(var[i].varValue-math.floor(var[i].varValue), 
361                             math.ceil(var[i].varValue)- var[i].varValue ) 
362                  if (frac> min_frac): 
363                      min_frac = frac 
364                      branching_var = i

366          else: 
367              print "Unknown branching strategy %s" %branch_strategy 
368              exit() 
369   
370          if branching_var >= 0: 
371              print "Branching on variable %s" %branching_var 
372   
373          #Create new nodes 
374          if search_strategy == DEPTH_FIRST: 
375              priority = -cur_depth - 1 
376          elif search_strategy == BEST_FIRST: 
377              priority = relax 
378          node_count += 1 
379          Q.push((node_count, cur_index, branching_var, '<=', math.floor(var[branching_var].varValue)), priority) 
380          node_count += 1

382          T.set_node_attr(cur_index, color, 'green')

406      if opt[i] > 0:
